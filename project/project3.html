TopBar 컴포넌트

1. TopBar 컴포넌트

Main 페이지 최상단에 로그인과 회원가입 페이지로 이동하기 위한 TopBar 컴포넌트가 존재해야한다.

구조는 App.js 파일에서 React Router를 사용하여 Main 페이지로 라우팅을 하며 Main 페이지의 TopBar 컴포넌트에는 로그인과 회원가입 메뉴가 있어 해당 페이지로 라우팅을 하는 구조로 pages 폴더에 Main.js 파일이 있으며 component 폴더에 TopBar.js 파일이 있다.

code

src/component/TopBar.js 

import "../App.css";
import { Link } from "react-router-dom"
                    
function TopBar() {
  return (
    <div className="header__topbar">
      <div className="container">
        <ul className="header__topbar__menu">
          <li><Link to="/login">로그인</Link></li>
          <li><Link to="/signup">회원가입</Link></li>
        </ul>
      </div>
    </div>
  );
}
                    
export default TopBar;

src/pages/Main.js 

import TopBar from "../component/TopBar";

function Main() {
  return (
    <header>
      <TopBar />
    </header>
  );
}

export default Main;

src/App.js 

import './App.css';
import { BrowserRouter, Routes, Route } from "react-router-dom";
import Main from "./pages/Main";

function App() {
  return (
    <BrowserRouter>
      <Routes>
          <Route path="/" element={<Main />} />
      </Routes>
    </BrowserRouter>
  );
}

export default App;
